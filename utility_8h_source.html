135  const unsigned int
136  m = (unsigned int)elem_mat.rows(),

138  libmesh_assert_equal_to(assembled_mat.rows(), m*2);

140  for (unsigned int i=0; i<m; i++)

144  assembled_mat(i,j+n) += -std::imag(elem_mat(i,j));
